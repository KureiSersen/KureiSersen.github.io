def segmentation(input_dir,output_dir):
    # pkuseg.test(input_dir, output_dir, nthread=28)   

    fo = open(output_dir,"r")
    lines = []
    for line in fo:
        # print(line)

        line_temp = line.split(" ")
        words = []
        for i in line_temp:
            i = re.sub(r'[^\w\s]','',i)
            i = i.strip()
            if len(i) >0:
                words.append(i)
        if len(words) > 0:
            lines.append(words)
            # print(words)
    return lines

if __name__ == '__main__':
